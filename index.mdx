    # Get an example query
    query <- get_example_query(model_id = "gem-1-bulk")$example_query

    # Generate a dataset
    result <- predict_query(query, model_id = "gem-1-bulk")

    # Access the results
    metadata <- result$metadata
    expression <- result$expression
